def get_data():
    """ 
    Function to download zip files, read them into dataframes, and save them to csv files. 

    The function loops through an array of file names and uses them to download the zip files one by one from the sepcified URLs. 
    The function then reads the contents of the zip file, opens the csv file inside, and converts it to a dataframe.
    The dataframe is then saved as a csv file on the local storage.
    """
    
    file_names = ['202001','202002', '202003']
    i=1
    for file in file_names:
        re = requests.get(f'https://s3.amazonaws.com/baywheels-data/{file}-baywheels-tripdata.csv.zip')
        zip_file = zipfile.ZipFile(io.BytesIO(re.content))
        csvfile = zip_file.open(f'{file}-baywheels-tripdata.csv')
        df = pd.read_csv(csvfile)
        df.to_csv(f'Ford0{i}-2020.csv',index=False) # save to csv file excluding the index column
        i +=1
